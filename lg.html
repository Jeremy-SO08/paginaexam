Carrera de Buses

Programa básico para simular una carrera de buses.

                import random
import time
import os

class Bus:
    def __init__(self, nombre, simbolo):
        self.nombre = nombre
        self.posicion = 0
        self.simbolo = simbolo

    def avanzar(self):
        avance = random.randint(1, 4)
        self.posicion += avance


def limpiar_pantalla(): 
    os.system('cls' if os.name == 'nt' else 'clear')



def mostrar_pista(bus1, bus2, meta):
    pista1 = " " * bus1.posicion + bus1.simbolo
    pista2 = " " * bus2.posicion + bus2.simbolo

    print(f"{bus1.nombre}:")
    print("|" + pista1)
    print("-" * (meta + 5))

    print(f"{bus2.nombre}:")
    print("|" + pista2)
    print("-" * (meta + 5))


def carrera():
    meta = 50

    bus1 = Bus("Bus Rojo", "🚌")
    bus2 = Bus("Bus Azul", "🚌")

   

    print("🏁 ¡Carrera de buses!\n")
    
    input("Presiona ENTER para comenzar...")

    while bus1.posicion < meta and bus2.posicion < meta:
        limpiar_pantalla()

        bus1.avanzar()
        bus2.avanzar()

        mostrar_pista(bus1, bus2, meta)

        time.sleep(0.5)

    print("\n🏆 RESULTADO FINAL 🏆")

    if bus1.posicion >= meta and bus2.posicion >= meta:
        print("🤝 ¡Empate!")
    elif bus1.posicion >= meta:
        print(f"🎉 ¡Ganó {bus1.nombre}!")
    else:
        print(f"🎉 ¡Ganó {bus2.nombre}!")


# Ejecutar
carrera()
limpiar_pantalla()
